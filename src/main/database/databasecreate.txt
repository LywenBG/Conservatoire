CREATE TABLE DISCIPLINE (
    DISNUM INT(50) NOT NULL PRIMARY KEY,
    DISLIBELLE CHAR(50)
);

CREATE TABLE ELEVE (
    ELENUM INT(50) NOT NULL PRIMARY KEY,
    DISNUM INT(50) NOT NULL,
    ELENOM CHAR(50),
    ELEPRENOM CHAR(50),
    ELECYCLE INT(1),
    ELEANNEECYCLE INT(1),
    ELELOGIN CHAR(50),
    ELEMDP CHAR(50)
);

CREATE TABLE PARTITIONELEVE (
    ELENUM INT(50) NOT NULL,
    PARNUM INT(50) NOT NULL,
    NUMEROPAGECLASSEUR INT(50),
    PRIMARY KEY (ELENUM, PARNUM)
);

CREATE TABLE PARTITIONS (
    PARNUM INT(50) NOT NULL PRIMARY KEY,
    PARNOM CHAR(50),
    PARAUTEUR CHAR(50)
);
